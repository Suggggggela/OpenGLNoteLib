

```cpp
void APIENTRY OpenGLDebugCallback(GLenum source, GLenum type, GLuint id, GLenum severity, GLsizei length, const GLchar* message, const void* userParam) {
    std::cerr << "OpenGL Debug Message: " << message << std::endl;
    std::cerr << "Source: " << source << ", Type: " << type << ", Severity: " << severity << std::endl;
}

void EnableOpenGLDebugging() {
    glEnable(GL_DEBUG_OUTPUT);  // 启用 OpenGL 调试输出
    glEnable(GL_DEBUG_OUTPUT_SYNCHRONOUS);  // 启用同步模式（阻塞等待错误）
    glDebugMessageCallback(OpenGLDebugCallback, nullptr);  // 设置回调函数
    glDebugMessageControl(GL_DONT_CARE, GL_DONT_CARE, GL_DONT_CARE, 0, nullptr, GL_TRUE);  // 启用所有类型的调试信息
}

```

Use:
```cpp
void main()
{
   EnableOpenGLDebugging(); 
   while(rendering)
{
    ...
}
}
```
