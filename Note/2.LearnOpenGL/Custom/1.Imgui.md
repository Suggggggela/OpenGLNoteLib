## 3.使用ImGUI

```cpp
//Imgui
ImGui::CreateContext();
ImGui_ImplGlfwGL3_Init(window, true);
ImGui::StyleColorsDark();
bool show_demo_window = true;
bool show_another_window = false;

glEnable(GL_DEPTH_TEST);

while(Rendering)
{
    //imgui
    ImGui_ImplGlfwGL3_NewFrame();
    ImGui::Begin("ImGui");
    ImGui::Text("Hello !");
    ImGui::SliderFloat3("m_ModelTransform", &modelTrans.x, -10.0f, 10.0f);
    ImGui::End();
    
    //imgui
    ImGui::Render();
    ImGui_ImplGlfwGL3_RenderDrawData(ImGui::GetDrawData());
}

//imgui
ImGui_ImplGlfwGL3_Shutdown();
ImGui::DestroyContext();


```

## 常用的ImGUI Element

Style
```cpp
ImGuiStyle& style = ImGui::GetStyle();

// 基础透明度
style.Alpha = 1.0f; // 控件整体透明度

// 窗口相关
style.WindowPadding = ImVec2(8, 8); // 窗口内边距
style.WindowRounding = 6.0f;        // 窗口圆角
style.WindowBorderSize = 1.0f;      // 窗口边框宽度

// 框架相关
style.FramePadding = ImVec2(4, 3);  // 控件（如按钮、输入框）内边距
style.FrameRounding = 3.0f;         // 控件圆角
style.FrameBorderSize = 0.0f;       // 控件边框宽度

// 项目相关
style.ItemSpacing = ImVec2(8, 4);   // 控件之间的间距
style.ItemInnerSpacing = ImVec2(4, 4); // 同一控件内部元素间距

// 滚动条相关
style.ScrollbarSize = 16.0f;        // 滚动条宽度
style.ScrollbarRounding = 9.0f;     // 滚动条圆角

// 字体相关
style.IndentSpacing = 21.0f;        // 缩进宽度
style.FontGlobalScale = 1.0f;       // 全局字体缩放

// 颜色相关（通过 ImGui::GetStyle().Colors[] 设置）
style.Colors[ImGuiCol_WindowBg] = ImVec4(0.1f, 0.1f, 0.1f, 1.0f); // 窗口背景色
style.Colors[ImGuiCol_Button]   = ImVec4(0.2f, 0.5f, 0.7f, 1.0f); // 按钮颜色
style.Colors[ImGuiCol_Text]     = ImVec4(1.0f, 1.0f, 1.0f, 1.0f); // 文本颜色

//UI字体大小
ImGui::CreateContext();
ImGuiIO& io = ImGui::GetIO(); 
ImFontConfig font_cfg;
font_cfg.SizePixels = 24.0f;
io.Fonts->AddFontDefault(&font_cfg);
```
